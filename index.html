// ================================================================= // 零用金記帳系統 · Google Apps Script 後端 // 版本:4.2 // - 讀取時帶上每筆紀錄的 row 編號(讓前端排序可以「同日內最新在最上面」) // - 刪除改用 row 編號定位,但仍驗證內容一致(雙重保險,避免併發誤刪) // ================================================================= const SHEET_LEDGER = '零用金收支'; const CACHE_LEDGER_TTL = 15; // ============= 取得工作表 ============= function getLedgerSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_LEDGER); if (!sheet) sheet = ss.getSheets()[0]; return sheet; } // ============= 快取工具 ============= function getCache() { return CacheService.getScriptCache(); } function clearLedgerCache() { try { getCache().remove('all_ledger'); } catch (e) {} } // ============= GET 請求(讀取,JSONP)============= function doGet(e) { const action = e.parameter.action; const callback = e.parameter.callback; let result; try { if (action === 'ping') { result = { status: 'ok', msg: '連線成功' }; } else if (action === 'getInit') { const date = e.parameter.date || ''; const all = readAllLedger(); result = { status: 'ok', allRecords: all, lunchRecords: date ? extractLunchFromLedger(all, date) : [] }; } else if (action === 'getAll') { result = { status: 'ok', records: readAllLedger() }; } else if (action === 'getLunchByDate') { const date = e.parameter.date; result = { status: 'ok', records: extractLunchFromLedger(readAllLedger(), date) }; } else if (action === 'getByDate') { const date = e.parameter.date; result = { status: 'ok', records: readAllLedger().filter(r => r.date === date) }; } else { result = { status: 'error', msg: '未知的動作: ' + action }; } } catch (err) { result = { status: 'error', msg: err.toString() }; } if (callback) { return ContentService .createTextOutput(callback + '(' + JSON.stringify(result) + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } // ============= POST 請求(寫入)============= function doPost(e) { try { const data = JSON.parse(e.postData.contents); if (data.action === 'append') { appendToLedger(getLedgerSheet(), data.record); clearLedgerCache(); return jsonResponse({ status: 'ok' }); } if (data.action === 'appendBatch') { const sheet = getLedgerSheet(); (data.records || []).forEach(r => appendToLedger(sheet, r)); clearLedgerCache(); return jsonResponse({ status: 'ok', count: (data.records || []).length }); } if (data.action === 'submitLunch') { const ledgerSheet = getLedgerSheet(); (data.ledger || []).forEach(r => appendToLedger(ledgerSheet, r)); clearLedgerCache(); return jsonResponse({ status: 'ok', ledgerCount: (data.ledger || []).length }); } if (data.action === 'deleteRecords') { const result = deleteRecords(data.targets || []); clearLedgerCache(); return jsonResponse(result); } return jsonResponse({ status: 'error', msg: '未知的動作' }); } catch (err) { return jsonResponse({ status: 'error', msg: err.toString() }); } } // ============= 寫入主表 ============= function appendToLedger(sheet, r) { sheet.appendRow([ r.date || '', r.item || '', r.type || '', r.income === '' ? '' : Number(r.income), r.expense === '' ? '' : Number(r.expense), r.handler || '', r.note || '', '' ]); } // ============= 刪除記錄(用 row 編號定位 + 內容驗證)============= function deleteRecords(targets) { if (!targets.length) { return { status: 'error', msg: '沒有要刪除的項目' }; } const sheet = getLedgerSheet(); const lastRow = sheet.getLastRow(); if (lastRow < 2) { return { status: 'error', msg: '主表沒有資料' }; } const toDelete = []; const mismatch = []; for (const target of targets) { const rowNum = Number(target.row || 0); if (rowNum < 2 || rowNum > lastRow) { mismatch.push(target.item || '(行號無效)'); continue; } const row = sheet.getRange(rowNum, 1, 1, 6).getValues()[0]; const rowDate = formatDate(row[0]); const rowItem = String(row[1] || ''); const rowType = String(row[2] || ''); const rowIncome = row[3] === '' ? '' : Number(row[3]); const rowExpense = row[4] === '' ? '' : Number(row[4]); const rowHandler = String(row[5] || ''); const tIncome = target.income === '' || target.income == null ? '' : Number(target.income); const tExpense = target.expense === '' || target.expense == null ? '' : Number(target.expense); const match = rowDate === String(target.date || '') && rowItem === String(target.item || '') && rowType === String(target.type || '') && rowIncome === tIncome && rowExpense === tExpense && rowHandler === String(target.handler || ''); if (match) { toDelete.push(rowNum); } else { mismatch.push(target.item || '(內容不符)'); } } toDelete.sort((a, b) => b - a); toDelete.forEach(rowNum => sheet.deleteRow(rowNum)); return { status: 'ok', deleted: toDelete.length, mismatch: mismatch, msg: '已刪除 ' + toDelete.length + ' 筆' + (mismatch.length ? ',有 ' + mismatch.length + ' 筆內容不符未刪除' : '') }; } // ============= 讀取主表全部(含快取,帶 row 編號)============= function readAllLedger() { const cache = getCache(); const cached = cache.get('all_ledger'); if (cached) { try { return JSON.parse(cached); } catch (e) {} } const sheet = getLedgerSheet(); const lastRow = sheet.getLastRow(); if (lastRow < 2) return []; const values = sheet.getRange(2, 1, lastRow - 1, 6).getValues(); const result = values.map((row, idx) => ({ row: idx + 2, // 試算表實際 row 編號(2-based,含表頭) date: formatDate(row[0]), item: row[1] || '', type: row[2] || '', income: row[3] === '' ? '' : Number(row[3]), expense: row[4] === '' ? '' : Number(row[4]), handler: row[5] || '' })).filter(r => r.item); try { const json = JSON.stringify(result); if (json.length < 95 * 1024) { cache.put('all_ledger', json, CACHE_LEDGER_TTL); } } catch (e) {} return result; } // ============= 從主表反解析便當明細 ============= function extractLunchFromLedger(allRecords, dateStr) { if (!dateStr) return []; const out = []; allRecords.forEach(r => { if (r.date !== dateStr) return; if (r.type !== '支出') return; const item = String(r.item || ''); const bentoMatch = item.match(/^(.+?)\s*[—–-]\s*(.+?)\s*[—–-]\s*便當\s*$/); if (bentoMatch) { const person = bentoMatch[1].trim(); let bento = bentoMatch[2].trim(); let lessRice = ''; const lr = bento.match(/^(.*?)\s*[((]\s*少飯\s*[))]\s*$/); if (lr) { bento = lr[1].trim(); lessRice = '少飯'; } out.push({ date: r.date, person, bento, lessRice, price: Number(r.expense || 0), type: '便當', handler: r.handler || '' }); return; } if (item.indexOf('午餐補助') !== -1) { const subMatch = item.match(/^(.+?)\s*[—–-]\s*午餐補助/); const person = subMatch ? subMatch[1].trim() : ''; out.push({ date: r.date, person, bento: '午餐補助', lessRice: '', price: Number(r.expense || 0), type: '補貼', handler: r.handler || '' }); } }); return out; } // ============= 日期格式化 ============= function formatDate(val) { if (!val && val !== 0) return ''; if (val instanceof Date) { if (isNaN(val.getTime())) return ''; const y = val.getFullYear(); const m = String(val.getMonth() + 1).padStart(2, '0'); const d = String(val.getDate()).padStart(2, '0'); return y + '-' + m + '-' + d; } const s = String(val).trim(); if (!s) return ''; if (/^\d{4}-\d{2}-\d{2}$/.test(s)) return s; const m1 = s.match(/^(\d{4})[\/\-](\d{1,2})[\/\-](\d{1,2})/); if (m1) { return m1[1] + '-' + String(m1[2]).padStart(2, '0') + '-' + String(m1[3]).padStart(2, '0'); } try { const d = new Date(s); if (!isNaN(d.getTime())) { const y = d.getFullYear(); const m = String(d.getMonth() + 1).padStart(2, '0'); const dd = String(d.getDate()).padStart(2, '0'); return y + '-' + m + '-' + dd; } } catch (e) {} return s; } // ============= 工具 ============= function jsonResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function manualClearAllCache() { getCache().remove('all_ledger'); Logger.log('快取已清除'); }
